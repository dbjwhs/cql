13 class DocumentFactory {
14 public:
15 // Creates a document based on file extension
16 static DocumentPtr createDocument(const std::string& filename) {
17 auto extension = std::filesystem::path(filename).extension().string();
18
19 // Convert to lowercase for case-insensitive comparison
20 std::transform(extension.begin(), extension.end(), extension.begin(),
21 [](unsigned char c) { return std::tolower(c); });
22
23 if (extension == ".pdf") {
24 return createPdfDocument(filename);
25 } else if (extension == ".docx" || extension == ".doc") {
26 return createWordDocument(filename);
27 } else if (extension == ".txt") {
28 return createTextDocument(filename);
29 } else {
30 throw std::invalid_argument(
31 std::format("Unsupported file extension: {}", extension));
32 }
33 }
