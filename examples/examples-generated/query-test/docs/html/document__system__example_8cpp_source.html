53 // Register with singleton document manager

57
58 // Create compatible tools using abstract factory
59 auto [viewer, editor, converter] = ToolFactory::createToolsFor(doc->getType());

62 // Apply strategy pattern
63 doc->applyFormatting("corporate");
64
65 // Observer pattern
66 doc->attachObserver(std::make_shared<AutoSaveObserver>());
67 doc->attachObserver(std::make_shared<ValidationObserver>());
68 doc->attachObserver(std::make_shared<LoggingObserver>());
69 std::cout << "Attached observers to document" << std::endl;
70
71 // Update content to trigger observers
72 std::cout << "\nUpdating document content..." << std::endl;
73 doc->updateContent("Updated content for the quarterly report.");
74
75 // Use tools
76 std::cout << "\nUsing document tools..." << std::endl;
77 viewer->view(doc);
78 editor->edit(doc);
79
80 // Using message queue for processing
81 std::cout << "\nSending document for processing via message queue..." << std::endl;
82 messageQueue.publish("document.process", doc);
83
84 // Wait for processing to complete
85 std::this_thread::sleep_for(std::chrono::seconds(2));
86
87 // Clean up
88 std::cout << "\nShutting down services..." << std::endl;
