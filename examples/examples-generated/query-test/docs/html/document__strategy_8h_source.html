106 // Register a formatting strategy factory
107 template<typename T>
108 void registerFormattingStrategy(const std::string& name) {
109 static_assert(std::is_base_of<TextFormattingStrategy, T>::value,
110 "Type must derive from TextFormattingStrategy");
111
112 formattingStrategies_[name] = []() {
113 return std::make_unique<T>();
114 };
115 }
116
117 // Get a formatting strategy by name
118 std::unique_ptr<TextFormattingStrategy> createFormattingStrategy(const std::string& name) {
119 auto it = formattingStrategies_.find(name);
120 if (it != formattingStrategies_.end()) {
121 return it->second();
122 }
123 throw std::invalid_argument("Unknown formatting strategy: " + name);
124 }
