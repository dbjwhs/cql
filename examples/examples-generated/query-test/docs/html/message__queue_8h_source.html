21// Message structure
22struct Message {
23 std::string topic;
24 std::any payload;
25 std::type_index payloadType;
26
27 template<typename T>
28 Message(std::string t, T&& p)
29 : topic(std::move(t)),
30 payload(std::forward<T>(p)),
31 payloadType(typeid(T)) {}
32
33 // Helper to get typed payload
34 template<typename T>
35 T getPayload() const {
36 if (payloadType != typeid(T)) {
37 throw std::runtime_error("Type mismatch in message payload");
38 }
39 return std::any_cast<T>(payload);
40 }
41};

43// Message queue for microservice communication
44class MessageQueue {
45public:
46 // Singleton access
47 static MessageQueue& getInstance() {
48 static MessageQueue instance;
49 return instance;
50 }
51
52 // Non-copyable and non-movable
53 MessageQueue(const MessageQueue&) = delete;
54 MessageQueue& operator=(const MessageQueue&) = delete;
55 MessageQueue(MessageQueue&&) = delete;
56 MessageQueue& operator=(MessageQueue&&) = delete;
57
58 // Publish a message to a topic
59 template<typename T>
60 void publish(const std::string& topic, T&& payload) {
61 {
62 std::lock_guard<std::mutex> lock(queueMutex_);
63 messageQueue_.push(Message(topic, std::forward<T>(payload)));
64 }
65
66 queueCondition_.notify_one();
67 }
68
69 // Subscribe to a topic with a callback
70 template<typename T>
71 void subscribe(const std::string& topic, std::function<void(T)> callback) {
72 std::lock_guard<std::mutex> lock(subscriptionMutex_);
73
74 auto wrappedCallback = [callback](const Message& msg) {
75 if (msg.payloadType == typeid(T)) {
76 callback(msg.getPayload<T>());
77 }
78 };
79
80 subscriptions_[topic].push_back(wrappedCallback);
81 }
82
83 // Start processing messages in the background
84 void startProcessing() {
85 if (processingThread_.joinable()) {
86 return; // Already running
87 }
88
89 running_ = true;
90 processingThread_ = std::thread([this]() {
91 this->processMessages();
92 });
93 }
94
95 // Stop processing messages
96 void stopProcessing() {
97 running_ = false;
98 queueCondition_.notify_all();
99
100 if (processingThread_.joinable()) {
101 processingThread_.join();
102 }
103 }
104
105 // Wait until the queue is empty
106 void waitUntilEmpty() {
107 std::unique_lock<std::mutex> lock(queueMutex_);
108 emptyCondition_.wait(lock, [this]() {
109 return messageQueue_.empty();
110 });
111 }
112
113 // Get the current queue size
114 size_t getQueueSize() const {
115 std::lock_guard<std::mutex> lock(queueMutex_);
116 return messageQueue_.size();
117 }
118
119 ~MessageQueue() {
120 stopProcessing();
121 }
122
123private:
124 MessageQueue() : running_(false) {}
125
126 // Process messages in the queue
127 void processMessages() {
128 while (running_) {
129 Message currentMessage = waitForMessage();
130
131 if (!running_) {
132 break;
133 }
134
135 dispatchMessage(currentMessage);
136
137 // Notify if the queue is empty
138 {
139 std::lock_guard<std::mutex> lock(queueMutex_);
140 if (messageQueue_.empty()) {
141 emptyCondition_.notify_all();
142 }
143 }
144 }
145 }
146
147 // Wait for a message to be available
148 Message waitForMessage() {
149 std::unique_lock<std::mutex> lock(queueMutex_);
150
151 queueCondition_.wait(lock, [this]() {
152 return !messageQueue_.empty() || !running_;
153 });
154
155 if (!running_) {
156 return Message("", 0); // Dummy message
157 }
158
159 Message message = messageQueue_.front();
160 messageQueue_.pop();
161 return message;
162 }
163
164 // Dispatch a message to all subscribers
165 void dispatchMessage(const Message& message) {
166 std::vector<std::function<void(const Message&)>> callbacks;
167
168 {
169 std::lock_guard<std::mutex> lock(subscriptionMutex_);
170 auto it = subscriptions_.find(message.topic);
171 if (it != subscriptions_.end()) {
172 callbacks = it->second;
173 }
174 }
175
176 for (const auto& callback : callbacks) {
177 callback(message);
178 }
179 }
180
181 mutable std::mutex queueMutex_;
182 mutable std::mutex subscriptionMutex_;
183 std::condition_variable queueCondition_;
184 std::condition_variable emptyCondition_;
185 std::queue<Message> messageQueue_;
186 std::unordered_map<std::string, std::vector<std::function<void(const Message&)>>> subscriptions_;
187 std::thread processingThread_;
188 std::atomic<bool> running_;
189};

191// Microservice base class
192class Microservice {
193public:
194 Microservice(const std::string& name) : name_(name), running_(false) {}
195
196 virtual ~Microservice() {
197 stop();
198 }
199
200 // Start the microservice
201 void start() {
202 if (running_) {
203 return;
204 }
205
206 running_ = true;
207
208 // Start message processing
209 MessageQueue::getInstance().startProcessing();
210
211 // Initialize the service
212 initialize();
213
214 // Start the service thread
215 serviceThread_ = std::thread([this]() {
216 this->run();
217 });
218 }
219
220 // Stop the microservice
221 void stop() {
222 running_ = false;
223
224 if (serviceThread_.joinable()) {
225 serviceThread_.join();
226 }
227 }
228
229 // Get the service name
230 std::string getName() const {
231 return name_;
232 }
233
234protected:
235 // Initialize subscriptions and service state
236 virtual void initialize() = 0;
237
238 // Main service loop
239 virtual void run() = 0;
240
241 // Helper for subscribing to messages
242 template<typename T>
243 void subscribe(const std::string& topic, std::function<void(T)> callback) {
244 MessageQueue::getInstance().subscribe<T>(topic, std::move(callback));
245 }
246
247 // Helper for publishing messages

249 void publish(const std::string& topic, T&& payload) {
250 MessageQueue::getInstance().publish(topic, std::forward<T>(payload));
251 }

253 std::string name_;
254 std::atomic<bool> running_;
