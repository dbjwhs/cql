26// Base document implementation with observer pattern support
27class BaseDocument : public Document {
28public:
29 explicit BaseDocument(const std::string& filename) : filename_(filename) {}
30
31 // Observer pattern implementation
32 void attachObserver(ObserverPtr observer) override {
33 if (observer) {
34 auto it = std::find(observers_.begin(), observers_.end(), observer);
35 if (it == observers_.end()) {
36 observers_.push_back(observer);
37 }
38 }
39 }
40
41 void detachObserver(ObserverPtr observer) override {
42 observers_.erase(
43 std::remove(observers_.begin(), observers_.end(), observer),
44 observers_.end()
45 );
46 }
47
48 void notifyObservers(const std::string& eventType) override {
49 for (const auto& observer : observers_) {
50 observer->onNotify(*this, eventType);
51 }
52 }
53
54 // Implement common methods from Document interface
55 std::string getFilename() const override {
56 return filename_;
57 }
58
59 void updateContent(const std::string& newContent) override {
60 content_ = newContent;
61 notifyObservers(events::DocumentChanged);
62 }
63
64 void setHeader(const std::string& header) override {
65 header_ = header;
66 }
67
68 void setContent(const std::string& content) override {
69 content_ = content;
70 }
71
72 void setFooter(const std::string& footer) override {
73 footer_ = footer;
74 }
75
76 void setMetadata(const MetadataMap& metadata) override {
77 metadata_ = metadata;
78 }
79
80protected:
81 std::string filename_;
82 std::string header_;
83 std::string content_;
84 std::string footer_;
85 MetadataMap metadata_;
86 std::vector<ObserverPtr> observers_;
87};
