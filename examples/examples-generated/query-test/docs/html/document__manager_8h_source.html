29 // Register a document in the manager
30 void registerDocument(DocumentPtr document) {
31 if (!document) {
32 throw std::invalid_argument("Cannot register a null document");
33 }
34
35 std::lock_guard<std::mutex> lock(mutex_);
36 const std::string& filename = document->getFilename();
37 documents_[filename] = document;
38 }
39
40 // Retrieve a document by filename
41 DocumentPtr getDocument(const std::string& filename) {
42 std::lock_guard<std::mutex> lock(mutex_);
43 auto it = documents_.find(filename);
44 if (it != documents_.end()) {
45 return it->second;
46 }
47 return nullptr;
48 }
49
50 // Remove a document from the manager
51 bool unregisterDocument(const std::string& filename) {
52 std::lock_guard<std::mutex> lock(mutex_);
53 return documents_.erase(filename) > 0;
54 }
55
56 // Get all registered documents
57 std::vector<DocumentPtr> getAllDocuments() {
58 std::lock_guard<std::mutex> lock(mutex_);
59 std::vector<DocumentPtr> result;
60 result.reserve(documents_.size());
61
62 for (const auto& [_, doc] : documents_) {
63 result.push_back(doc);
64 }
65
66 return result;
67 }
68
69 // Get count of registered documents
70 size_t getDocumentCount() const {
71 std::lock_guard<std::mutex> lock(mutex_);
72 return documents_.size();
73 }
74
75private:
76 // Private constructor for singleton
77 DocumentManager() = default;
78
79 // Thread synchronization
80 mutable std::mutex mutex_;
81
82 // Document storage
83 std::unordered_map<std::string, DocumentPtr> documents_;
84};
