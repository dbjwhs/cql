72 // Create a temporary output directory for testing

111 // Format it as a proper Claude API response
112 return create_mock_claude_response(mock_response_content);

114 // Return an error for an empty query

131 config.set_output_directory(output_dir);
132 config.set_overwrite_existing_files(true);
133
134 // Create an ApiClient pointed at our mock server
135 ApiClient client(config);
136

181 // Save the files to disk

184 TEST_ASSERT(saved, "File should be saved successfully");
185 }
186
187 // Check if the files exist on disk
188 for (const auto& file : files) {
189 std::string filepath = output_dir + "/" + file.m_filename;
190 TEST_ASSERT(std::filesystem::exists(filepath),
191 "Generated file should exist on disk: " + filepath);
192 }
193
194 // Stop the mock server
195 server.stop();
196
197 // Clean up the test output directory
198 for (const auto& entry : std::filesystem::directory_iterator(output_dir)) {
199 std::filesystem::remove(entry.path());
200 }
201
202 return TestResult::pass();
203 } catch (const std::exception& e) {
204 return TestResult::fail("Exception in test_api_integration: " + std::string(e.what()),
205 __FILE__, __LINE__);
