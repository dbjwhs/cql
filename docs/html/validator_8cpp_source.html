12// helper function to get token type for a node
13TokenType get_node_type(const QueryNode* node) {
14 if (dynamic_cast<const CodeRequestNode*>(node)) {
15 // coderequestnode represents both language and description
16 // for validation, we need to count it as both
17 return TokenType::LANGUAGE; // we'll handle description separately
18 }
19 if (dynamic_cast<const ContextNode*>(node)) return TokenType::CONTEXT;
20 if (dynamic_cast<const TestNode*>(node)) return TokenType::TEST;
21 if (dynamic_cast<const DependencyNode*>(node)) return TokenType::DEPENDENCY;
22 if (dynamic_cast<const PerformanceNode*>(node)) return TokenType::PERFORMANCE;
23 if (dynamic_cast<const CopyrightNode*>(node)) return TokenType::COPYRIGHT;
24 if (dynamic_cast<const ArchitectureNode*>(node)) return TokenType::ARCHITECTURE;
25 if (dynamic_cast<const ConstraintNode*>(node)) return TokenType::CONSTRAINT;
26 if (dynamic_cast<const ExampleNode*>(node)) return TokenType::EXAMPLE;
27 if (dynamic_cast<const SecurityNode*>(node)) return TokenType::SECURITY;
28 if (dynamic_cast<const ComplexityNode*>(node)) return TokenType::COMPLEXITY;
29 if (dynamic_cast<const ModelNode*>(node)) return TokenType::MODEL;
30 if (dynamic_cast<const FormatNode*>(node)) return TokenType::FORMAT;
31 if (dynamic_cast<const VariableNode*>(node)) return TokenType::VARIABLE;
32
33 // default fallback
34 return TokenType::IDENTIFIER;
35}

80 // Add a custom rule to validate architecture pattern compatibility
81 add_custom_rule([](const std::vector<std::unique_ptr<QueryNode>>& nodes) -> std::optional<ValidationIssue> {
82 // Collect all architecture nodes
83 std::vector<const ArchitectureNode*> architecture_nodes;
84 for (const auto& node : nodes) {
85 if (auto* arch_node = dynamic_cast<const ArchitectureNode*>(node.get())) {
86 architecture_nodes.push_back(arch_node);
87 }
88 }
89
90 // Skip validation if fewer than 2 architecture patterns
91 if (architecture_nodes.size() < 2) {
92 return std::nullopt;
93 }
94
95 // Check pattern compatibility
96 PatternCompatibilityManager compatibility_manager;
97 auto issues = compatibility_manager.check_compatibility(architecture_nodes);
98
99 // Return a validation issue if incompatibilities are found
100 if (!issues.empty()) {
101 std::string message = "Architecture pattern compatibility issues found: ";
102 for (size_t i = 0; i < issues.size(); ++i) {
103 if (i > 0) message += "; ";
104 message += issues[i].to_string();
105 }
106
107 return ValidationIssue(
108 ValidationSeverity::WARNING, // Use WARNING to provide guidance but not prevent code generation
109 message
110 );
111 }
112
113 return std::nullopt;
114 });
115}

117std::map<TokenType, int> QueryValidator::count_directives(const std::vector<std::unique_ptr<QueryNode>>& nodes) {
118 std::map<TokenType, int> counts;
119
120 for (const auto& node : nodes) {
121 // special handling for coderequestnode which contains both language and description
122 if (auto* code_node = dynamic_cast<const CodeRequestNode*>(node.get())) {
123 // Only count language if it's not empty
124 if (!code_node->language().empty()) {
125 counts[TokenType::LANGUAGE]++;
126 }
127 counts[TokenType::DESCRIPTION]++;
128 } else {
129 TokenType type = get_node_type(node.get());
130 counts[type]++;
131 }
132 }
133
134 return counts;
135}

167std::vector<ValidationIssue> QueryValidator::check_dependencies(const std::map<TokenType, int>& counts) {
168 std::vector<ValidationIssue> issues;
169
170 for (const auto& [dependent, dependency] : m_dependency_rules) {
171 if (counts.find(dependent) != counts.end() && counts.at(dependent) > 0) {
172 if (counts.find(dependency) == counts.end() || counts.at(dependency) == 0) {
173 issues.emplace_back(
174 ValidationSeverity::WARNING,
175 "Directive @" + token_type_to_string(dependent) +
176 " works best with @" + token_type_to_string(dependency) + "."
177 );
178 }
179 }
180 }
181
182 return issues;

230std::vector<ValidationIssue> QueryValidator::validate(const std::vector<std::unique_ptr<QueryNode>>& nodes) {
231 std::vector<ValidationIssue> issues;
232
233 // Count directive occurrences - this handles empty nodes gracefully
234 std::map<TokenType, int> counts = count_directives(nodes);
235
236 // Run all validation checks
237 auto required_issues = check_required(counts);
238 auto exclusive_issues = check_exclusive(counts);
239 auto dependency_issues = check_dependencies(counts);
240 auto incompatibility_issues = check_incompatibilities(counts);
241 auto custom_issues = run_custom_rules(nodes);
242
243 // Combine all issues, prioritizing required directive issues first
244 issues.insert(issues.end(), required_issues.begin(), required_issues.end());
245 issues.insert(issues.end(), exclusive_issues.begin(), exclusive_issues.end());
246 issues.insert(issues.end(), dependency_issues.begin(), dependency_issues.end());
247 issues.insert(issues.end(), incompatibility_issues.begin(), incompatibility_issues.end());
248 issues.insert(issues.end(), custom_issues.begin(), custom_issues.end());
249
250 // Throw ValidationException if there are any ERROR issues
251 // We collect all errors first instead of throwing on the first one
252 std::vector<std::string> error_messages;
253 for (const auto& issue : issues) {
254 if (issue.severity == ValidationSeverity::ERROR) {
255 error_messages.push_back(issue.message);
256 }
257 }
258
259 if (!error_messages.empty()) {
260 // Join all error messages into a single error message
261 std::string combined_message;
262 for (size_t ndx = 0; ndx < error_messages.size(); ++ndx) {
263 if (ndx > 0) {
264 combined_message += "; ";
265 }
266 combined_message += error_messages[ndx];
267 }
268
269 // Determine the appropriate error code
270 std::string error_code = validation_errors::GENERAL_ERROR;
271
272 // Use the MULTIPLE_ERRORS code if there are multiple errors
273 if (error_messages.size() > 1) {
274 error_code = validation_errors::MULTIPLE_ERRORS;
275 }
276 // Otherwise, use a specific code based on the directive that's missing
277 else if (combined_message.find("Required directive @LANGUAGE") != std::string::npos) {
278 error_code = validation_errors::MISSING_LANGUAGE;
279 }
280 else if (combined_message.find("Required directive @DESCRIPTION") != std::string::npos) {
281 error_code = validation_errors::MISSING_DESCRIPTION;
282 }
283 else if (combined_message.find("Required directive @COPYRIGHT") != std::string::npos) {
284 error_code = validation_errors::MISSING_COPYRIGHT;
285 }
286
287 throw ValidationException(combined_message, error_code);
288 }
289
290 return issues;
291}
