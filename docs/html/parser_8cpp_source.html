30std::vector<std::unique_ptr<QueryNode>> Parser::parse() {
31 std::vector<std::unique_ptr<QueryNode>> nodes;
32
33 while (m_current_token) {
34 if (m_current_token->m_type == TokenType::NEWLINE) {
35 advance();
36 continue;
37 }
38
39 // validate the token is a keyword
40 if (m_current_token->m_type != TokenType::LANGUAGE &&
41 m_current_token->m_type != TokenType::DESCRIPTION &&
42 m_current_token->m_type != TokenType::CONTEXT &&
43 m_current_token->m_type != TokenType::TEST &&
44 m_current_token->m_type != TokenType::DEPENDENCY &&
45 m_current_token->m_type != TokenType::PERFORMANCE &&
46 m_current_token->m_type != TokenType::COPYRIGHT &&
47 // phase 2 directives
48 m_current_token->m_type != TokenType::ARCHITECTURE &&
49 m_current_token->m_type != TokenType::CONSTRAINT &&
50 m_current_token->m_type != TokenType::EXAMPLE &&
51 m_current_token->m_type != TokenType::SECURITY &&
52 m_current_token->m_type != TokenType::COMPLEXITY &&
53 m_current_token->m_type != TokenType::MODEL &&
54 m_current_token->m_type != TokenType::FORMAT &&
55 m_current_token->m_type != TokenType::VARIABLE) {
56
57 if (!m_current_token) {
58 throw ParserError("Expected keyword but reached end of input",
59 m_lexer.current_line(), m_lexer.current_column());
60 } else {
61 throw ParserError("Expected keyword",
62 m_current_token->m_line, m_current_token->m_column);
63 }
64 }
65
66 // parse the appropriate node type
67 switch (m_current_token->m_type) {
68 case TokenType::LANGUAGE:
69 nodes.push_back(parse_code_request());
70 break;
71 case TokenType::CONTEXT:
72 nodes.push_back(parse_context());
73 break;
74 case TokenType::TEST:
75 nodes.push_back(parse_test());
76 break;
77 case TokenType::DEPENDENCY:
78 nodes.push_back(parse_dependency());
79 break;
80 case TokenType::PERFORMANCE:
81 nodes.push_back(parse_performance());
82 break;
83 case TokenType::COPYRIGHT:
84 nodes.push_back(parse_copyright());
85 break;
86 // phase 2 directives
87 case TokenType::ARCHITECTURE:
88 nodes.push_back(parse_architecture());
89 break;
90 case TokenType::CONSTRAINT:
91 nodes.push_back(parse_constraint());
92 break;
93 case TokenType::EXAMPLE:
94 nodes.push_back(parse_example());
95 break;
96 case TokenType::SECURITY:
97 nodes.push_back(parse_security());
98 break;
99 case TokenType::COMPLEXITY:
100 nodes.push_back(parse_complexity());
101 break;
102 case TokenType::MODEL:
103 nodes.push_back(parse_model());
104 break;
105 case TokenType::FORMAT:
106 nodes.push_back(parse_format());
107 break;
108 case TokenType::VARIABLE:
109 nodes.push_back(parse_variable());
110 break;
111 default:
112 throw ParserError("Unexpected token type",
113 m_current_token->m_line, m_current_token->m_column);
114 }
115 }
116
117 return nodes;
118}

120std::string Parser::parse_string() {
121 // skip whitespace tokens
122 while (m_current_token && m_current_token->m_type == TokenType::NEWLINE) {
123 advance();
124 }
125
126 if (!m_current_token) {
127 throw ParserError("Unexpected end of input while expecting string",
128 m_lexer.current_line(), m_lexer.current_column());
129 }
130
131 if (m_current_token->m_type != TokenType::STRING) {
132 throw ParserError(
133 "Expected string (got " + token_type_to_string(m_current_token->m_type) +
134 " with value '" + m_current_token->m_value + "')",
135 m_current_token->m_line, m_current_token->m_column
136 );
137 }
138
139 std::string value = m_current_token->m_value;
140 advance(); // move to the next token
141 return value;
142}

144std::unique_ptr<QueryNode> Parser::parse_code_request() {
145 size_t line = m_current_token->m_line;
146 size_t column = m_current_token->m_column;
147
148 advance(); // skip past @language token
149
150 // get the language string
151 std::string language = parse_string();
152
153 // skip any newlines
154 while (m_current_token && m_current_token->m_type == TokenType::NEWLINE) {
155 advance();
156 }
157
158 // now expect @description
159 if (!m_current_token || m_current_token->m_type != TokenType::DESCRIPTION) {
160 throw ParserError(
161 "Expected @description after @language",
162 m_current_token ? m_current_token->m_line : line,
163 m_current_token ? m_current_token->m_column : column
164 );
165 }
166
167 advance(); // skip past @description token
168 std::string description = parse_string();
169
170 return std::make_unique<CodeRequestNode>(language, description);
171}

212std::unique_ptr<QueryNode> Parser::parse_architecture() {
213 advance(); // skip @architecture
214
215 // Backup the current token to restore if needed
216 std::optional<Token> saved_token = m_current_token;
217
218 // Check for layered format - first token should be the layer type
219 std::string first_token;
220 if (m_current_token && m_current_token->m_type == TokenType::IDENTIFIER) {
221 first_token = m_current_token->m_value;
222 advance(); // consume the layer identifier
223
224 // Convert to lowercase for consistent checking
225 std::string lowercase_layer = first_token;
226 std::transform(lowercase_layer.begin(), lowercase_layer.end(), lowercase_layer.begin(),
227 [](unsigned char c) { return std::tolower(c); });
228
229 // Check if this is a valid layer
230 if (lowercase_layer == "foundation" || lowercase_layer == "component" || lowercase_layer == "interaction") {
231 // This is the layered format
232 PatternLayer layer = string_to_pattern_layer(lowercase_layer);
233
234 // Next token should be the pattern name
235 std::string pattern_name = parse_string();
236
237 // Check if we have parameters
238 std::string parameters;
239 if (m_current_token && m_current_token->m_type == TokenType::STRING) {
240 parameters = parse_string();
241 }
242
243 return std::make_unique<ArchitectureNode>(layer, pattern_name, parameters);
244 }
245
246 // If we got here, it wasn't a valid layer
247 // Restore token and continue with the standard format
248 m_current_token = saved_token;
249
250 // We'll just reparse the string as the regular format
251 // The token position is already saved and restored above
252 }
253
254 // Old format - just a single string
255 std::string architecture = parse_string();
256 return std::make_unique<ArchitectureNode>(architecture);
257}
