11// claude query language (cql)
12//
13// history:
14// the claude query language was developed in 2025 as a domain-specific language to formalize
15// and standardize how developers craft queries for large language models (llms), specifically
16// anthropic's claude. it follows the compiler pattern where a high-level representation (the cql)
17// is translated into a more detailed and structured query string.
18//
19// purpose:
20// cql addresses several challenges in prompt engineering:
21// 1. consistency - providing a standard structure for technical queries
22// 2. efficiency - reducing time spent crafting detailed prompts manually
23// 3. quality - ensuring all necessary information is included for optimal code generation
24// 4. reusability - allowing query templates to be saved, modified, and reused
25//
26// common usage patterns:
27// 1. specification of technical requirements:
28// @language "c++"
29// @description "implement a thread-safe queue with timeout features"
30//
31// 2. providing implementation context:
32// @context "designed for a real-time system with strict latency requirements"
33// @context "must be compatible with c++20 features"
34//
35// 3. test case specifications:
36// @test "empty queue behavior"
37// @test "concurrent push/pop operations"
38// @test "timeout handling for blocked operations"
39//
40// 4. extending with custom sections:
41// @dependencies "requires boost::asio for async operations"
42// @performance "must handle 10k operations per second with <1ms latency"
43//
44// 5. copyright and license information:
45// @copyright "mit license" "2025 dbjwhs"
46

56// visitor pattern for different node types
57class QueryVisitor {
58public:
59 virtual ~QueryVisitor() = default;
60 virtual void visit(const CodeRequestNode& node) = 0;

63 virtual void visit(const DependencyNode& node) = 0;

65 virtual void visit(const CopyrightNode& node) = 0;
66};

175// query compiler that generates structured prompts
176class QueryCompiler final : public QueryVisitor {
177public:
178 void visit(const CodeRequestNode& node) override {
179 m_result_sections["code"] = "Please generate " + node.language() + " code that:\n" +
180 node.description() + "\n\n";
181 }
182
183 void visit(const ContextNode& node) override {
184 if (!m_result_sections.count("context")) {
185 m_result_sections["context"] = "Context:\n";
186 }
187 m_result_sections["context"] += "- " + node.context() + "\n";
188 }
189
190 void visit(const TestNode& node) override {
191 // store the test cases, we'll format them when getting the final query
192 m_test_cases.insert(m_test_cases.end(),
193 node.test_cases().begin(),
194 node.test_cases().end());
195 }
196
197 void visit(const DependencyNode& node) override {
198 if (!m_result_sections.count("dependencies")) {
199 m_result_sections["dependencies"] = "Dependencies:\n";
200 }
201 for (const auto& dependency : node.dependencies()) {
202 m_result_sections["dependencies"] += "- " + dependency + "\n";
203 }
204 }
205
206 void visit(const PerformanceNode& node) override {
207 if (!m_result_sections.count("performance")) {
208 m_result_sections["performance"] = "Performance Requirements:\n";
209 }
210 m_result_sections["performance"] += "- " + node.requirement() + "\n";
211 }
212
213 void visit(const CopyrightNode& node) override {
214 std::string copyright_message = "Please include the following copyright header at the top of all generated files:\n";
215 copyright_message += "```\n";
216 copyright_message += "// " + node.license() + "\n";
217 copyright_message += "// Copyright (c) " + node.owner() + "\n";
218 copyright_message += "```\n\n";
219 m_result_sections["copyright"] = copyright_message;
220 }
221
222 [[nodiscard]] std::string get_compiled_query() const {
223 std::string query_string;
224
225 // add a copyright section if it exists
226 auto copyright_it = m_result_sections.find("copyright");
227 if (copyright_it != m_result_sections.end()) {
228 query_string += copyright_it->second;
229 }
230
231 // add a code section if it exists
232 auto code_it = m_result_sections.find("code");
233 if (code_it != m_result_sections.end()) {
234 query_string += code_it->second;
235 }
236
237 // add a context section if it exists
238 auto context_it = m_result_sections.find("context");
239 if (context_it != m_result_sections.end()) {
240 query_string += context_it->second + "\n";
241 }
242
243 // add a dependencies section if it exists
244 auto dependencies_it = m_result_sections.find("dependencies");
245 if (dependencies_it != m_result_sections.end()) {
246 query_string += dependencies_it->second + "\n";
247 }
248
249 // add a performance section if it exists
250 auto performance_it = m_result_sections.find("performance");
251 if (performance_it != m_result_sections.end()) {
252 query_string += performance_it->second + "\n";
253 }
254
255 // add test cases if we have any
256 if (!m_test_cases.empty()) {
257 query_string += "Please include tests for the following cases:\n";
258 for (const auto& test_case : m_test_cases) {
259 query_string += "- " + test_case + "\n";
260 }
261 query_string += "\n";
262 }
263
264 // add a quality assurance section as a standard footer
265 query_string += "Quality Assurance Requirements:\n";
266 query_string += "- All code must be well-documented with comments\n";
267 query_string += "- Follow modern C++ best practices\n";
268 query_string += "- Ensure proper error handling\n";
269 query_string += "- Optimize for readability and maintainability\n";
270
271 return query_string;
272 }
273
274 void print_compiled_query(std::ostream& out = std::cout) const {
275 out << "\n=== Compiled Query ===\n\n"
276 << get_compiled_query()
277 << "===================\n";
278 }
279
280private:
281 std::map<std::string, std::string> m_result_sections;
282 std::vector<std::string> m_test_cases;
283};

285// token types for our dsl
286enum class TokenType {
287 LANGUAGE, // @language
288 DESCRIPTION, // @description
289 CONTEXT, // @context
290 TEST, // @test
291 DEPENDENCY, // @dependency
292 PERFORMANCE, // @performance
293 COPYRIGHT, // @copyright
294 IDENTIFIER, // any text
295 STRING, // "quoted text"
296 NEWLINE, // \n
297 END // end of input
298};
299
300// utility function to convert tokentype to string
301std::string token_type_to_string(TokenType type) {
302 switch (type) {
303 case TokenType::LANGUAGE: return "LANGUAGE";
304 case TokenType::DESCRIPTION: return "DESCRIPTION";
305 case TokenType::CONTEXT: return "CONTEXT";
306 case TokenType::TEST: return "TEST";
307 case TokenType::DEPENDENCY: return "DEPENDENCY";
308 case TokenType::PERFORMANCE: return "PERFORMANCE";
309 case TokenType::COPYRIGHT: return "COPYRIGHT";
310 case TokenType::IDENTIFIER: return "IDENTIFIER";
311 case TokenType::STRING: return "STRING";
312 case TokenType::NEWLINE: return "NEWLINE";
313 case TokenType::END: return "END";
314 default: return "UNKNOWN";
315 }
316}

318// token structure for lexical analysis
319struct Token {
320 TokenType m_type;
321 std::string m_value;
322 size_t m_line;
323 size_t m_column;
324
325 Token(TokenType t, std::string v, size_t l, size_t c)
326 : m_type(t), m_value(std::move(v)), m_line(l), m_column(c) {}
327
328 [[nodiscard]] std::string to_string() const {
329 std::ostringstream oss;
330 oss << "Token{type=" << token_type_to_string(m_type)
331 << ", value='" << m_value << "'"
332 << ", line=" << m_line
333 << ", column=" << m_column << "}";
334 return oss.str();
335 }
336};

338// lexical analyzer (lexer) for tokenizing input
339class Lexer {
340public:
341 explicit Lexer(std::string_view input)
342 : m_input(input), m_current(0), m_line(1), m_column(1) {}
343
344 std::optional<Token> next_token() {
345 skip_whitespace();
346
347 if (m_current >= m_input.length()) {
348 return std::nullopt;
349 }
350
351 if (m_input[m_current] == '@') {
352 return lex_keyword();
353 }
354
355 if (m_input[m_current] == '"') {
356 return lex_string();
357 }
358
359 if (m_input[m_current] == '\n') {
360 auto token = Token(TokenType::NEWLINE, "\n", m_line, m_column);
361 advance();
362 m_line++;
363 m_column = 1;
364 return token;
365 }
366
367 return lex_identifier();
368 }
369
370private:
371 std::string_view m_input;
372 size_t m_current;
373 size_t m_line;
374 size_t m_column;
375
376 // move to the next character in the input
377 void advance() {
378 m_current++;
379 m_column++;
380 }
381
382 // skip whitespace characters (except newlines)
383 void skip_whitespace() {
384 while (m_current < m_input.length() &&
385 (std::isspace(m_input[m_current]) && m_input[m_current] != '\n')) {
386 advance();
387 }
388 }
389
390 // parse a keyword token (starting with @)
391 std::optional<Token> lex_keyword() {
392 advance(); // skip @
393 std::string keyword;
394 size_t start_column = m_column;
395
396 while (m_current < m_input.length() && std::isalpha(m_input[m_current])) {
397 keyword += m_input[m_current];
398 advance();
399 }
400
401 TokenType type;
402 if (keyword == "language") type = TokenType::LANGUAGE;
403 else if (keyword == "description") type = TokenType::DESCRIPTION;
404 else if (keyword == "context") type = TokenType::CONTEXT;
405 else if (keyword == "test") type = TokenType::TEST;
406 else if (keyword == "dependency") type = TokenType::DEPENDENCY;
407 else if (keyword == "performance") type = TokenType::PERFORMANCE;
408 else if (keyword == "copyright") type = TokenType::COPYRIGHT;
409 else throw std::runtime_error("Unknown keyword: @" + keyword);
410
411 return Token(type, keyword, m_line, start_column);
412 }
413
414 // parse a string token (enclosed in quotes)
415 std::optional<Token> lex_string() {
416 if (m_current >= m_input.length() || m_input[m_current] != '"') {
417 throw std::runtime_error("Expected opening quote at line " +
418 std::to_string(m_line) + ", column " + std::to_string(m_column));
419 }
420
421 advance(); // skip opening quote
422 std::string value;
423 const size_t start_column = m_column;
424
425 while (m_current < m_input.length() && m_input[m_current] != '"') {
426 if (m_input[m_current] == '\\') {
427 advance();
428 if (m_current >= m_input.length()) {
429 throw std::runtime_error("Unterminated string escape sequence");
430 }
431 switch (m_input[m_current]) {
432 case 'n': value += '\n'; break;
433 case 't': value += '\t'; break;
434 case '"': value += '"'; break;
435 case '\\': value += '\\'; break;
436 default: throw std::runtime_error("Invalid escape sequence");
437 }
438 } else {
439 value += m_input[m_current];
440 }
441 advance();
442 }
443
444 if (m_current >= m_input.length()) {
445 throw std::runtime_error("Unterminated string");
446 }
447
448 advance(); // skip closing quote
449 return Token(TokenType::STRING, value, m_line, start_column);
450 }
451
452 // parse an identifier token
453 std::optional<Token> lex_identifier() {
454 std::string value;
455 const size_t start_column = m_column;
456
457 while (m_current < m_input.length() &&
458 !std::isspace(m_input[m_current]) &&
459 m_input[m_current] != '@') {
460 value += m_input[m_current];
461 advance();
462 }
463
464 return Token(TokenType::IDENTIFIER, value, m_line, start_column);
465 }
466};

468// parser for building the abstract syntax tree (ast)
469class Parser {
470public:
471 explicit Parser(const std::string_view input) : m_lexer(input) {
472 advance();
473 }
474
475 std::vector<std::unique_ptr<QueryNode>> parse() {
476 std::vector<std::unique_ptr<QueryNode>> nodes;
477
478 while (m_current_token) {
479 if (m_current_token->m_type == TokenType::NEWLINE) {
480 advance();
481 continue;
482 }
483
484 // validate the token is a keyword
485 if (m_current_token->m_type != TokenType::LANGUAGE &&
486 m_current_token->m_type != TokenType::DESCRIPTION &&
487 m_current_token->m_type != TokenType::CONTEXT &&
488 m_current_token->m_type != TokenType::TEST &&
489 m_current_token->m_type != TokenType::DEPENDENCY &&
490 m_current_token->m_type != TokenType::PERFORMANCE &&
491 m_current_token->m_type != TokenType::COPYRIGHT) {
492 throw std::runtime_error("Expected keyword at line " +
493 std::to_string(m_current_token->m_line));
494 }
495
496 // parse the appropriate node type
497 switch (m_current_token->m_type) {
498 case TokenType::LANGUAGE:
499 nodes.push_back(parse_code_request());
500 break;
501 case TokenType::CONTEXT:
502 nodes.push_back(parse_context());
503 break;
504 case TokenType::TEST:
505 nodes.push_back(parse_test());
506 break;
507 case TokenType::DEPENDENCY:
508 nodes.push_back(parse_dependency());
509 break;
510 case TokenType::PERFORMANCE:
511 nodes.push_back(parse_performance());
512 break;
513 case TokenType::COPYRIGHT:
514 nodes.push_back(parse_copyright());
515 break;
516 default:
517 throw std::runtime_error("Unexpected token type");
518 }
519 }
520
521 return nodes;
522 }
523
524private:
525 Lexer m_lexer;
526 std::optional<Token> m_current_token;
527
528 void advance() {
529 m_current_token = m_lexer.next_token();
530 }
531
532 // parse a string token
533 std::string parse_string() {
534 // skip whitespace tokens
535 while (m_current_token && m_current_token->m_type == TokenType::NEWLINE) {
536 advance();
537 }
538
539 if (!m_current_token) {
540 throw std::runtime_error("Unexpected end of input while expecting string");
541 }
542
543 if (m_current_token->m_type != TokenType::STRING) {
544 throw std::runtime_error(
545 "Expected string at line " + std::to_string(m_current_token->m_line) +
546 ", column " + std::to_string(m_current_token->m_column) +
547 " (got " + token_type_to_string(m_current_token->m_type) +
548 " with value '" + m_current_token->m_value + "')"
549 );
550 }
551
552 std::string value = m_current_token->m_value;
553 advance(); // move to the next token
554 return value;
555 }
556
557 // parse a code request node (@language + @description)
558 std::unique_ptr<QueryNode> parse_code_request() {
559 advance(); // skip past @language token
560
561 // get the language string
562 std::string language = parse_string();
563
564 // skip any newlines
565 while (m_current_token && m_current_token->m_type == TokenType::NEWLINE) {
566 advance();
567 }
568
569 // now expect @description
570 if (!m_current_token || m_current_token->m_type != TokenType::DESCRIPTION) {
571 throw std::runtime_error(
572 "Expected @description after @language at line " +
573 std::to_string(m_current_token ? m_current_token->m_line : 0) +
574 (m_current_token ? " (got '" + m_current_token->m_value + "')" : "")
575 );
576 }
577
578 advance(); // skip past @description token
579 std::string description = parse_string();
580
581 return std::make_unique<CodeRequestNode>(language, description);
582 }
583
584 // parse a context node
585 std::unique_ptr<QueryNode> parse_context() {
586 advance(); // skip @context
587 std::string context = parse_string();
588 return std::make_unique<ContextNode>(context);
589 }
590
591 // parse a test node
592 std::unique_ptr<QueryNode> parse_test() {
593 advance(); // skip @test
594 std::vector<std::string> test_cases;
595
596 // get the first test case
597 test_cases.push_back(parse_string());
598
599 return std::make_unique<TestNode>(std::move(test_cases));
600 }
601
602 // parse a dependency node
603 std::unique_ptr<QueryNode> parse_dependency() {
604 advance(); // skip @dependency
605 std::vector<std::string> dependencies;
606
607 // get the dependency
608 dependencies.push_back(parse_string());
609
610 return std::make_unique<DependencyNode>(std::move(dependencies));
611 }
612
613 // parse a performance node
614 std::unique_ptr<QueryNode> parse_performance() {
615 advance(); // skip @performance
616 std::string requirement = parse_string();
617 return std::make_unique<PerformanceNode>(requirement);
618 }
619
620 // parse a copyright node
621 std::unique_ptr<QueryNode> parse_copyright() {
622 advance(); // skip @copyright
623 std::string license = parse_string();
624 std::string owner = parse_string();
625 return std::make_unique<CopyrightNode>(license, owner);
626 }
627};
