94 bool all_passed = true;
95

105 {"Validation Requirements", test_validation_requirements},
106 {"Phase 2 Features", test_phase2_features},

119 {"Lexer (Standalone)", test_lexer_standalone},
120 {"Parser (Standalone)", test_parser_standalone},

123

125 for (const auto& test : tests) {

139 print_test_result(test.name, result);
140 all_passed = false;
141
142 if (fail_fast) {
143 std::cout << "\nFailed fast: Stopping tests after first exception." << std::endl;
144 break;
145 }
146 }
147 }
148
149 // print summary

151 std::cout << "\n\033[32mAll tests passed!\033[0m" << std::endl;
152 } else {
153 std::cout << "\n\033[31mSome tests failed!\033[0m" << std::endl;
