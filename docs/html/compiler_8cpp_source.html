98// process a string with variable interpolation (${var})
99std::string QueryCompiler::interpolate_variables(const std::string& input) const {
100 std::string result = input;
101 size_t start_pos = 0;
102
103 // find all occurrences of ${variable_name} and replace them
104 while ((start_pos = result.find("${", start_pos)) != std::string::npos) {
105 size_t end_pos = result.find("}", start_pos);
106 if (end_pos == std::string::npos) {
107 break; // no closing bracket, stop processing
108 }
109
110 // extract the variable name
111 std::string var_name = result.substr(start_pos + 2, end_pos - start_pos - 2);
112
113 // look up the variable value
114 auto it = m_variables.find(var_name);
115 if (it != m_variables.end()) {
116 // replace the variable reference with its value
117 result.replace(start_pos, end_pos - start_pos + 1, it->second);
118 // move past the replacement
119 start_pos += it->second.length();
120 } else {
121 // variable not found, leave it as is and move past it
122 start_pos = end_pos + 1;
123 }
124 }
125
126 return result;
127}

129std::string QueryCompiler::get_compiled_query() const {
130 std::string query_string;
131
132 // add model-specific preamble if not using the default model
133 if (m_target_model != "claude-3-opus") {
134 query_string += "Target Model: " + m_target_model + "\n\n";
135 }
136
137 // add a copyright section if it exists
138 auto copyright_it = m_result_sections.find("copyright");
139 if (copyright_it != m_result_sections.end()) {
140 query_string += copyright_it->second;
141 }
142
143 // add a code section if it exists
144 auto code_it = m_result_sections.find("code");
145 if (code_it != m_result_sections.end()) {
146 query_string += code_it->second;
147 }
148
149 // add a context section if it exists
150 auto context_it = m_result_sections.find("context");
151 if (context_it != m_result_sections.end()) {
152 query_string += context_it->second + "\n";
153 }
154
155 // add an architecture section if it exists
156 auto architecture_it = m_result_sections.find("architecture");
157 if (architecture_it != m_result_sections.end()) {
158 query_string += architecture_it->second + "\n";
159 }
160
161 // add a constraints section if it exists
162 auto constraints_it = m_result_sections.find("constraints");
163 if (constraints_it != m_result_sections.end()) {
164 query_string += constraints_it->second + "\n";
165 }
166
167 // add a dependencies section if it exists
168 auto dependencies_it = m_result_sections.find("dependencies");
169 if (dependencies_it != m_result_sections.end()) {
170 query_string += dependencies_it->second + "\n";
171 }
172
173 // add a performance section if it exists
174 auto performance_it = m_result_sections.find("performance");
175 if (performance_it != m_result_sections.end()) {
176 query_string += performance_it->second + "\n";
177 }
178
179 // add a security section if it exists
180 auto security_it = m_result_sections.find("security");
181 if (security_it != m_result_sections.end()) {
182 query_string += security_it->second + "\n";
183 }
184
185 // add a complexity section if it exists
186 auto complexity_it = m_result_sections.find("complexity");
187 if (complexity_it != m_result_sections.end()) {
188 query_string += complexity_it->second + "\n";
189 }
190
191 // add code examples if we have any
192 if (!m_examples.empty()) {
193 query_string += "Please reference these examples:\n";
194 for (const auto& [label, code] : m_examples) {
195 query_string += "Example - " + label + ":\n";
196 query_string += "```\n" + code + "\n```\n\n";
197 }
198 }
199
200 // add test cases if we have any
201 if (!m_test_cases.empty()) {
202 query_string += "Please include tests for the following cases:\n";
203 for (const auto& test_case : m_test_cases) {
204 query_string += "- " + test_case + "\n";
205 }
206 query_string += "\n";
207 }
208
209 // add a quality assurance section as a standard footer
210 query_string += "Quality Assurance Requirements:\n";
211 query_string += "- All code must be well-documented with comments\n";
212 query_string += "- Follow modern C++ best practices\n";
213 query_string += "- Ensure proper error handling\n";
214 query_string += "- Optimize for readability and maintainability\n";
215
216 // format the output appropriately
217 if (m_output_format == "json") {
218 // convert to json format (simplified)
219 std::string json_output = "{\n";
220 json_output += " \"query\": " + std::string(query_string.empty() ? "\"\"" : "\"" + query_string + "\"") + ",\n";
221 json_output += " \"model\": \"" + m_target_model + "\",\n";
222 json_output += " \"format\": \"" + m_output_format + "\"\n";
223 json_output += "}\n";
224 return json_output;
225 }
226
227 // process template variables
228 if (!m_variables.empty()) {
229 query_string = interpolate_variables(query_string);
230 }
231
232 return query_string;
233}
