89// Pattern layer conversions
90std::string pattern_layer_to_string(PatternLayer layer) {
91 switch (layer) {
92 case PatternLayer::FOUNDATION: return "foundation";
93 case PatternLayer::COMPONENT: return "component";
94 case PatternLayer::INTERACTION: return "interaction";
95 default: return "unknown";
96 }
97}
98
99PatternLayer string_to_pattern_layer(const std::string& layer_str) {
100 // Convert to lowercase for consistent comparison
101 std::string lowercase_layer = layer_str;
102 std::transform(lowercase_layer.begin(), lowercase_layer.end(), lowercase_layer.begin(),
103 [](unsigned char c) { return std::tolower(c); });
104
105 if (lowercase_layer == "foundation") {
106 return PatternLayer::FOUNDATION;
107 } else if (lowercase_layer == "interaction") {
108 return PatternLayer::INTERACTION;
109 } else {
110 // Default to component for anything else
111 return PatternLayer::COMPONENT;
112 }
113}

119ArchitectureNode::ArchitectureNode(PatternLayer layer, std::string pattern_name, std::string parameters)
120 : m_layer(layer),
121 m_pattern_name(std::move(pattern_name)),
122 m_parameters(std::move(parameters)),
123 m_is_layered_format(true) {
124
125 // Also set the combined architecture string for backward compatibility
126 m_architecture = pattern_layer_to_string(m_layer) + " \"" + m_pattern_name + "\"";
127 if (!m_parameters.empty()) {

129 }
