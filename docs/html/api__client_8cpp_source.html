18struct ApiClient::Impl {
19 Config m_config;
20 ApiClientStatus m_status;
21 std::string m_last_error;
22 CURL* m_curl;
23 std::string m_response_buffer;
24 bool m_initialized;

26 explicit Impl(const Config& config)
27 : m_config(config),
28 m_status(ApiClientStatus::Ready),
29 m_curl(nullptr),
30 m_initialized(false) {
31
32 // Initialize CURL
33 curl_global_init(CURL_GLOBAL_ALL);
34 m_curl = curl_easy_init();
35
36 if (m_curl) {
37 m_initialized = true;
38 Logger::getInstance().log(LogLevel::INFO, "ApiClient initialized with model: ", config.get_model());
39 } else {
40 m_status = ApiClientStatus::Error;
41 m_last_error = "Failed to initialize CURL";
42 Logger::getInstance().log(LogLevel::ERROR, "Failed to initialize CURL");
43 }
44 }
45
46 ~Impl() {
47 // Clean up CURL
48 if (m_curl) {
49 curl_easy_cleanup(m_curl);
50 }
51 curl_global_cleanup();
52 }

54 // Callback function for CURL to write response data
55 static size_t write_callback(void* contents, size_t size, size_t nmemb, std::string* buffer) {
56 size_t real_size = size * nmemb;
57 buffer->append(static_cast<char*>(contents), real_size);
58 return real_size;
59 }

61 // Prepare the request for the Claude API
62 struct curl_slist* prepare_request(const std::string& query, std::string& request_data) {
63 // Construct the request JSON
64 nlohmann::json request_json = {
65 {"model", m_config.get_model()},
66 {"max_tokens", 100000},
67 {"messages", {
68 {
69 {"role", "user"},
70 {"content", query}
71 }
72 }}
73 };
74
75 request_data = request_json.dump();
76
77 // Construct the full URL by combining base URL and endpoint
78 std::string api_url = m_config.get_api_base_url() + "/v1/messages";
79
80 // Set up CURL request
81 curl_easy_setopt(m_curl, CURLOPT_URL, api_url.c_str());
82 curl_easy_setopt(m_curl, CURLOPT_WRITEFUNCTION, write_callback);
83 curl_easy_setopt(m_curl, CURLOPT_WRITEDATA, &m_response_buffer);
84 curl_easy_setopt(m_curl, CURLOPT_POSTFIELDS, request_data.c_str());
85 curl_easy_setopt(m_curl, CURLOPT_TIMEOUT, m_config.get_timeout());
86
87 // Set headers
88 struct curl_slist* headers = nullptr;
89 headers = curl_slist_append(headers, "Content-Type: application/json");
90 std::string auth_header = "x-api-key: " + m_config.get_api_key();
91 headers = curl_slist_append(headers, auth_header.c_str());
92 headers = curl_slist_append(headers, "anthropic-version: 2023-06-01");
93 curl_easy_setopt(m_curl, CURLOPT_HTTPHEADER, headers);
94
95 return headers;
96 }

98 // Execute the API request
99 CURLcode execute_request(struct curl_slist* headers) const {
100 // Perform the request
101 Logger::getInstance().log(LogLevel::INFO, "Sending request to Claude API...");
102 CURLcode res = curl_easy_perform(m_curl);
103
104 // Cleanup headers
105 curl_slist_free_all(headers);
106
107 return res;
108 }
109
110 // Process the API response
111 ApiResponse process_response(CURLcode curl_result, const ApiResponse& partial_response) {
112 ApiResponse response = partial_response;
113
114 // Check for CURL errors
115 if (curl_result != CURLE_OK) {
116 m_status = ApiClientStatus::Error;
117 m_last_error = curl_easy_strerror(curl_result);
118 response.m_error_message = m_last_error;
119 response.m_status_code = 0;
120 Logger::getInstance().log(LogLevel::ERROR, "CURL error: ", m_last_error);
121 return response;
122 }
123
124 // Get status code
125 long status_code;
126 curl_easy_getinfo(m_curl, CURLINFO_RESPONSE_CODE, &status_code);
127 response.m_status_code = static_cast<int>(status_code);
128
129 // Process response based on status code
130 if (status_code >= 200 && status_code < 300) {
131 return process_successful_response(response);
132 } else if (status_code == 429) {
133 return process_rate_limited_response(response);
134 } else {
135 return process_error_response(response, status_code);
136 }
137 }
138
139 // Process a successful API response (status code 2xx)
140 ApiResponse process_successful_response(ApiResponse& response) {
141 m_status = ApiClientStatus::Ready;
142 response.m_success = true;
143 response.m_raw_response = m_response_buffer;
144
145 try {
146 // Parse JSON response
147 nlohmann::json json_response = nlohmann::json::parse(m_response_buffer);
148 if (json_response.contains("content") && json_response["content"].is_array()) {
149 // Extract content from response
150 for (const auto& content : json_response["content"]) {
151 if (content.contains("text") && content["text"].is_string()) {
152 response.m_raw_response = content["text"];
153 break;
154 }
155 }
156 }
157 Logger::getInstance().log(LogLevel::INFO, "API request successful");
158 } catch (const std::exception& e) {
159 Logger::getInstance().log(LogLevel::ERROR, "Error parsing API response: ", e.what());
160 // Still mark as success since we got a valid HTTP response,
161 // The response processor will handle extracting code blocks
162 }
163
164 return response;
165 }
166
167 // Process a rate-limited API response (status code 429)
168 ApiResponse process_rate_limited_response(ApiResponse& response) {
169 m_status = ApiClientStatus::RateLimited;
170 response.m_error_message = "Rate limited: Too many requests";
171 Logger::getInstance().log(LogLevel::ERROR, "API rate limit exceeded (429)");
172 return response;
173 }
174
175 // Process an error API response (status code >= 400, except 429)
176 ApiResponse process_error_response(ApiResponse& response, long status_code) {
177 m_status = ApiClientStatus::Error;
178
179 try {
180 // Try to parse error from JSON
181 nlohmann::json json_response = nlohmann::json::parse(m_response_buffer);
182 if (json_response.contains("error") && json_response["error"].is_object()) {
183 auto error = json_response["error"];
184 if (error.contains("message") && error["message"].is_string()) {
185 m_last_error = error["message"];
186 }
187 }
188 } catch (...) {
189 // If parsing fails, use the raw response as error
190 m_last_error = "API error " + std::to_string(status_code) + ": " + m_response_buffer;
191 }
192
193 response.m_error_message = m_last_error;
194 Logger::getInstance().log(LogLevel::ERROR, "API error (", status_code, "): ", m_last_error);
195 return response;
196 }
197
198 // Send a request to the Claude API - main entry point
199 ApiResponse send_request(const std::string& query) {
200 if (!m_initialized) {
201 ApiResponse response;
202 response.m_success = false;
203 response.m_status_code = 0;
204 response.m_error_message = "CURL not initialized";
205 return response;
206 }
207
208 // Reset response buffer
209 m_response_buffer.clear();
210
211 // Set client status
212 m_status = ApiClientStatus::Processing;
213
214 // Prepare request
215 std::string request_data;
216 struct curl_slist* headers = prepare_request(query, request_data);
217
218 // Prepare response
219 ApiResponse response;
220 response.m_success = false;
221
222 // Execute request
223 CURLcode curl_result = execute_request(headers);
224
225 // Process response
226 return process_response(curl_result, response);
227 }
228};

298Config Config::load_from_default_locations() {
299 Config config;
300
301 // Try to load from environment variables first
302 if (const char *api_key_env = std::getenv("CQL_API_KEY")) {
303 config.m_api_key = api_key_env;
304 }
305
306 if (const char *model_env = std::getenv("CQL_MODEL")) {
307 config.m_model = model_env;
308 }
309
310 if (const char *base_url_env = std::getenv("CQL_API_BASE_URL")) {
311 config.m_api_base_url = base_url_env;
312 }
313
314 if (const char *timeout_env = std::getenv("CQL_TIMEOUT")) {
315 try {
316 config.m_timeout = std::stoi(timeout_env);
317 } catch (...) {
318 Logger::getInstance().log(LogLevel::ERROR, "Invalid timeout value in environment variable");
319 }
320 }
321
322 if (const char *max_retries_env = std::getenv("CQL_MAX_RETRIES")) {
323 try {
324 config.m_max_retries = std::stoi(max_retries_env);
325 } catch (...) {
326 Logger::getInstance().log(LogLevel::ERROR, "Invalid max_retries value in environment variable");
327 }
328 }
329
330 if (const char *output_dir_env = std::getenv("CQL_OUTPUT_DIR")) {
331 config.m_output_directory = output_dir_env;
332 }
333
334 // Then try to load from a config file
335 std::string home_dir;
336 if (const char *home_env = std::getenv("HOME")) {
337 home_dir = home_env;
338 } else {
339 home_dir = "."; // Fallback to current directory
340 }
341
342 std::string config_path = home_dir + "/.cql/config.json";
343
344 // Check if a config file exists
345 if (std::filesystem::exists(config_path)) {
346 try {
347 config = load_from_file(config_path);
348 } catch (const std::exception& e) {
349 Logger::getInstance().log(LogLevel::ERROR, "Error loading config file: ", e.what());
350 }
351 }
352
353 return config;
354}

356Config Config::load_from_file(const std::string& filename) {
357 Config config;
358
359 try {
360 // Read the file
361 std::ifstream file(filename);
362 if (!file.is_open()) {
363 throw std::runtime_error("Failed to open config file: " + filename);
364 }
365
366 // Parse JSON
367 nlohmann::json json_config;
368 file >> json_config;
369
370 // Extract values
371 if (json_config.contains("api") && json_config["api"].is_object()) {
372 auto api = json_config["api"];
373
374 if (api.contains("key") && api["key"].is_string()) {
375 config.m_api_key = api["key"];
376 }
377
378 if (api.contains("model") && api["model"].is_string()) {
379 config.m_model = api["model"];
380 }
381
382 if (api.contains("base_url") && api["base_url"].is_string()) {
383 config.m_api_base_url = api["base_url"];
384 }
385
386 if (api.contains("timeout") && api["timeout"].is_number()) {
387 config.m_timeout = api["timeout"];
388 }
389
390 if (api.contains("max_retries") && api["max_retries"].is_number()) {
391 config.m_max_retries = api["max_retries"];
392 }
393 }
394
395 if (json_config.contains("output") && json_config["output"].is_object()) {
396 auto output = json_config["output"];
397
398 if (output.contains("default_directory") && output["default_directory"].is_string()) {
399 config.m_output_directory = output["default_directory"];
400
401 // Replace ~ with home directory
402 if (!config.m_output_directory.empty() && config.m_output_directory[0] == '~') {
403 if (const char *home_env = std::getenv("HOME")) {
404 config.m_output_directory.replace(0, 1, home_env);
405 }
406 }
407 }
408
409 if (output.contains("create_missing_dirs") && output["create_missing_dirs"].is_boolean()) {
410 config.m_create_missing_dirs = output["create_missing_dirs"];
411 }
412
413 if (output.contains("overwrite_existing") && output["overwrite_existing"].is_boolean()) {
414 config.m_overwrite_existing = output["overwrite_existing"];
415 }
416 }
417
418 Logger::getInstance().log(LogLevel::INFO, "Loaded configuration from ", filename);
419 } catch (const std::exception& e) {
420 Logger::getInstance().log(LogLevel::ERROR, "Error parsing config file: ", e.what());
421 throw;
422 }
423
424 return config;
425}
