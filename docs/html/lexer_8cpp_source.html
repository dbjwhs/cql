64std::optional<Token> Lexer::next_token() {
65 skip_whitespace();
66
67 if (m_current >= m_input.length()) {
68 return std::nullopt;
69 }
70
71 if (m_input[m_current] == '@') {
72 return lex_keyword();
73 }
74
75 if (m_input[m_current] == '"') {
76 return lex_string();
77 }
78
79 if (m_input[m_current] == '\n') {
80 auto token = Token(TokenType::NEWLINE, "\n", m_line, m_column);
81 advance();
82 m_line++;
83 m_column = 1;
84 return token;
85 }
86
87 return lex_identifier();
88}

90std::optional<Token> Lexer::lex_keyword() {
91 advance(); // skip @
92 std::string keyword;
93 size_t start_column = m_column;
94
95 while (m_current < m_input.length() && std::isalpha(m_input[m_current])) {
96 keyword += m_input[m_current];
97 advance();
98 }
99
100 TokenType type;
101 if (keyword == "language") type = TokenType::LANGUAGE;
102 else if (keyword == "description") type = TokenType::DESCRIPTION;
103 else if (keyword == "context") type = TokenType::CONTEXT;
104 else if (keyword == "test") type = TokenType::TEST;
105 else if (keyword == "dependency") type = TokenType::DEPENDENCY;
106 else if (keyword == "performance") type = TokenType::PERFORMANCE;
107 else if (keyword == "copyright") type = TokenType::COPYRIGHT;
108 // New directive types for Phase 2
109 else if (keyword == "architecture") type = TokenType::ARCHITECTURE;
110 else if (keyword == "constraint") type = TokenType::CONSTRAINT;
111 else if (keyword == "example") type = TokenType::EXAMPLE;
112 else if (keyword == "security") type = TokenType::SECURITY;
113 else if (keyword == "complexity") type = TokenType::COMPLEXITY;
114 else if (keyword == "model") type = TokenType::MODEL;
115 else if (keyword == "format") type = TokenType::FORMAT;
116 else if (keyword == "variable") type = TokenType::VARIABLE;
117 else {
118 throw LexerError("Unknown keyword: @" + keyword, m_line, start_column - 1);
119 }
120
121 return Token(type, keyword, m_line, start_column);
122}

124std::optional<Token> Lexer::lex_string() {
125 if (m_current >= m_input.length() || m_input[m_current] != '"') {
126 throw LexerError("Expected opening quote", m_line, m_column);
127 }
128
129 advance(); // skip opening quote
130 std::string value;
131 const size_t start_column = m_column;
132
133 while (m_current < m_input.length() && m_input[m_current] != '"') {
134 if (m_input[m_current] == '\\') {
135 advance();
136 if (m_current >= m_input.length()) {
137 throw LexerError("Unterminated string escape sequence", m_line, m_column);
138 }
139 switch (m_input[m_current]) {
140 case 'n': value += '\n'; break;
141 case 't': value += '\t'; break;
142 case '"': value += '"'; break;
143 case '\\': value += '\\'; break;
144 default:
145 throw LexerError("Invalid escape sequence", m_line, m_column);
146 }
147 } else {
148 value += m_input[m_current];
149 }
150 advance();
151 }
152
153 if (m_current >= m_input.length()) {
154 throw LexerError("Unterminated string", m_line, start_column);
155 }
156
157 advance(); // skip closing quote
158 return Token(TokenType::STRING, value, m_line, start_column);
159}

161std::optional<Token> Lexer::lex_identifier() {
162 std::string value;
163 const size_t start_column = m_column;
164
165 while (m_current < m_input.length() &&
166 !std::isspace(m_input[m_current]) &&
167 m_input[m_current] != '@') {
168 value += m_input[m_current];
169 advance();
170 }
171
172 return Token(TokenType::IDENTIFIER, value, m_line, start_column);
173}
