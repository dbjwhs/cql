13TestResult test_architecture_patterns() {
14 std::cout << "Testing architecture patterns..." << std::endl;

16 try {
17 // Test pattern layer string conversion
18 TEST_ASSERT(pattern_layer_to_string(PatternLayer::FOUNDATION) == "foundation",
19 "FOUNDATION should convert to 'foundation'");
20 TEST_ASSERT(pattern_layer_to_string(PatternLayer::COMPONENT) == "component",
21 "COMPONENT should convert to 'component'");
22 TEST_ASSERT(pattern_layer_to_string(PatternLayer::INTERACTION) == "interaction",
23 "INTERACTION should convert to 'interaction'");
24
25 // Test string to pattern layer conversion
26 TEST_ASSERT(string_to_pattern_layer("foundation") == PatternLayer::FOUNDATION,
27 "'foundation' should convert to FOUNDATION");
28 TEST_ASSERT(string_to_pattern_layer("FOUNDATION") == PatternLayer::FOUNDATION,
29 "'FOUNDATION' should convert to FOUNDATION");
30 TEST_ASSERT(string_to_pattern_layer("component") == PatternLayer::COMPONENT,
31 "'component' should convert to COMPONENT");
32 TEST_ASSERT(string_to_pattern_layer("interaction") == PatternLayer::INTERACTION,
33 "'interaction' should convert to INTERACTION");
34
35 // Test default case
36 TEST_ASSERT(string_to_pattern_layer("unknown") == PatternLayer::COMPONENT,
37 "'unknown' should default to COMPONENT");
38
39 // Test pattern creation from nodes
40 ArchitectureNode foundation_node(PatternLayer::FOUNDATION, "microservices", "");
41 ArchitectureNode component_node1(PatternLayer::COMPONENT, "factory_method", "products: [\"Document\", \"Image\"]");
42 ArchitectureNode component_node2(PatternLayer::COMPONENT, "singleton", "thread_safe: true");
43 ArchitectureNode interaction_node(PatternLayer::INTERACTION, "observer", "events: [\"documentChanged\"]");
44
45 // Create patterns from nodes
46 Pattern foundation_pattern(foundation_node);
47 Pattern component_pattern1(component_node1);
48 Pattern component_pattern2(component_node2);
49 Pattern interaction_pattern(interaction_node);
50
51 // Verify pattern properties
52 TEST_ASSERT(foundation_pattern.get_layer() == PatternLayer::FOUNDATION,
53 "Foundation pattern should have FOUNDATION layer");
54 TEST_ASSERT(foundation_pattern.get_name() == "microservices",
55 "Foundation pattern name should be 'microservices'");
56
57 TEST_ASSERT(component_pattern1.get_layer() == PatternLayer::COMPONENT,
58 "Component pattern should have COMPONENT layer");
59 TEST_ASSERT(component_pattern1.get_name() == "factory_method",
60 "Component pattern name should be 'factory_method'");
61 TEST_ASSERT(component_pattern1.get_parameters().find("products") != std::string::npos,
62 "Component pattern parameters should contain 'products'");
63
64 TEST_ASSERT(interaction_pattern.get_layer() == PatternLayer::INTERACTION,
65 "Interaction pattern should have INTERACTION layer");
66 TEST_ASSERT(interaction_pattern.get_name() == "observer",
67 "Interaction pattern name should be 'observer'");
68
69 // Test pattern compatibility manager
70 PatternCompatibilityManager manager;
71
72 // Test compatible patterns
73 TEST_ASSERT(manager.are_patterns_compatible(component_pattern1, component_pattern2) == true,
74 "Factory method and singleton should be compatible");
75
76 // Test different layers (should always be compatible)
77 TEST_ASSERT(manager.are_patterns_compatible(foundation_pattern, component_pattern1) == true,
78 "Patterns in different layers should be compatible");
79 TEST_ASSERT(manager.are_patterns_compatible(interaction_pattern, component_pattern2) == true,
80 "Patterns in different layers should be compatible");
81
82 // Test foundation uniqueness constraint
83 ArchitectureNode foundation_node2(PatternLayer::FOUNDATION, "layered_architecture", "");
84 Pattern foundation_pattern2(foundation_node2);
85
86 std::vector<Pattern> patterns = {foundation_pattern, foundation_pattern2};
87 auto issues = manager.check_compatibility(patterns);
88
89 TEST_ASSERT(!issues.empty(),
90 "Multiple foundation patterns should generate compatibility issues");
91
92 // Test incompatible patterns
93 ArchitectureNode prototype_node(PatternLayer::COMPONENT, "prototype", "deep_copy: true");
94 Pattern prototype_pattern(prototype_node);
95
96 TEST_ASSERT(manager.are_patterns_compatible(component_pattern2, prototype_pattern) == false,
97 "Singleton and prototype should be incompatible");
98
99 // Test structural patterns
100 ArchitectureNode bridge_node(PatternLayer::COMPONENT, "bridge",
101 "implementors: [\"WindowsRenderer\", \"MacOSRenderer\"]");
102 ArchitectureNode composite_node(PatternLayer::COMPONENT, "composite",
103 "component_type: \"UIComponent\"");
104 ArchitectureNode decorator_node(PatternLayer::COMPONENT, "decorator",
105 "decorations: [\"Border\", \"Shadow\"]");
106
107 Pattern bridge_pattern(bridge_node);
108 Pattern composite_pattern(composite_node);
109 Pattern decorator_pattern(decorator_node);
110
111 // Test bridge and composite incompatibility
112 TEST_ASSERT(manager.are_patterns_compatible(bridge_pattern, composite_pattern) == false,
113 "Bridge and composite should be incompatible");
114
115 // Test decorator compatibility
116 TEST_ASSERT(manager.are_patterns_compatible(decorator_pattern, composite_pattern) == true,
117 "Decorator and composite should be compatible");
118
119 // Test multiple patterns compatibility
120 std::vector<Pattern> ui_patterns = {
121 composite_pattern, decorator_pattern, component_pattern1 // factory + composite + decorator
122 };
123 auto ui_issues = manager.check_compatibility(ui_patterns);
124 TEST_ASSERT(ui_issues.empty(),
125 "Factory + Composite + Decorator should be compatible");
126
127 // Test behavioral patterns
128 ArchitectureNode observer_node(PatternLayer::INTERACTION, "observer",

130 ArchitectureNode command_node(PatternLayer::INTERACTION, "command",

142 TEST_ASSERT(manager.are_patterns_compatible(command_pattern, strategy_pattern) == true,
143 "Command and Strategy should be compatible");
144
145 // Test combined creational, structural, and behavioral patterns
146 std::vector<Pattern> complex_patterns = {
147 component_pattern1, // factory_method - creational
148 decorator_pattern, // decorator - structural
149 observer_pattern // observer - behavioral
150 };
151
152 auto complex_issues = manager.check_compatibility(complex_patterns);
153 TEST_ASSERT(complex_issues.empty(),
154 "Factory + Decorator + Observer should be compatible");
155

157 ArchitectureNode legacy_node("Singleton pattern with thread safety");
158 Pattern legacy_pattern(legacy_node);
159
160 TEST_ASSERT(legacy_pattern.get_layer() == PatternLayer::COMPONENT,
161 "Legacy pattern should default to COMPONENT layer");
162
163 // Try using a compatibility issue's string representation
164 if (!issues.empty()) {
165 std::string issue_str = issues[0].to_string();
166 TEST_ASSERT(!issue_str.empty(),
167 "Compatibility issue string representation should not be empty");
168 }
169
170 return TestResult::pass();
171 } catch (const std::exception& e) {
172 return TestResult::fail("Exception in test_architecture_patterns: " +
173 std::string(e.what()), __FILE__, __LINE__);
174 }
175}
