35// extract regex matches from text using a pattern
36std::vector<std::vector<std::string>> extract_regex_matches(
37 const std::string& content,
38 const std::string& pattern,
39 size_t expected_groups
40) {
41 std::vector<std::vector<std::string>> results;
42 std::regex regex_pattern(pattern);
43
44 auto begin = std::sregex_iterator(content.begin(), content.end(), regex_pattern);
45 auto end = std::sregex_iterator();
46
47 for (std::sregex_iterator it = begin; it != end; ++it) {
48 std::smatch match = *it;
49
50 // skip if we don't have the expected number of groups
51 if (expected_groups > 0 && match.size() <= expected_groups) {
52 continue;
53 }
54
55 // each match has multiple groups
56 std::vector<std::string> groups;
57 for (size_t i = 0; i < match.size(); ++i) {
58 groups.push_back(match[i].str());
59 }
60
61 results.push_back(groups);
62 }
63
64 return results;
65}

67// extract string values that match a specific regex group
68std::set<std::string> extract_regex_group_values(
69 const std::string& content,
70 const std::string& pattern,
71 size_t group_index
72) {
73 std::set<std::string> values;
74 std::regex regex_pattern(pattern);
75
76 auto begin = std::sregex_iterator(content.begin(), content.end(), regex_pattern);
77 auto end = std::sregex_iterator();
78
79 for (std::sregex_iterator it = begin; it != end; ++it) {
80 std::smatch match = *it;
81 if (match.size() > group_index) {
82 values.insert(match[group_index].str());
83 }
84 }
85
86 return values;
87}

106std::string QueryProcessor::compile(const std::string_view query_str) {
107 // Store parser errors to report them after validation
108 std::optional<std::string> parser_error;
109 std::vector<std::unique_ptr<QueryNode>> nodes;
110
111 // Try to parse the query string, but catch parser errors
112 try {
113 Parser parser(query_str);
114 nodes = parser.parse();
115 } catch (const ParserError& e) {
116 // Store the error message but continue with validation on any nodes we did parse
117 parser_error = e.what();
118 } catch (const std::exception& e) {
119 // For other errors, just rethrow
120 throw;
121 }
122
123 // If no nodes were parsed at all, we can't validate, so just report the parser error
124 if (nodes.empty() && parser_error) {
125 throw std::runtime_error(*parser_error);
126 }
127
128 // Always attempt validation, even if parsing had errors
129 std::vector<ValidationIssue> validation_issues;
130 try {
131 // validate the query structure with whatever nodes we have
132 QueryValidator validator;
133 validation_issues = validator.validate(nodes);
134 } catch (const ValidationException& e) {
135 // For validation errors, use the formatted message which includes the error code
136 throw std::runtime_error(e.formatted_message());
137 }
138
139 // Report validation issues
140 for (const auto& issue : validation_issues) {
141 std::string level;
142 switch (issue.severity) {
143 case ValidationSeverity::INFO:
144 level = "INFO";
145 break;
146 case ValidationSeverity::WARNING:
147 level = "WARNING";
148 break;
149 case ValidationSeverity::ERROR:
150 level = "ERROR";
151 break;
152 }
153
154 Logger::getInstance().log(
155 issue.severity == ValidationSeverity::ERROR ? LogLevel::ERROR : LogLevel::NORMAL,
156 "Validation ", level, ": ", issue.message
157 );
158
159 // For errors, throw an exception - validation errors take precedence
160 if (issue.severity == ValidationSeverity::ERROR) {
161 // Create a ValidationException with a specific error code based on the message
162 std::string error_code = validation_errors::GENERAL_ERROR;
163
164 if (issue.message.find("Required directive @LANGUAGE") != std::string::npos) {
165 error_code = validation_errors::MISSING_LANGUAGE;
166 }
167 else if (issue.message.find("Required directive @DESCRIPTION") != std::string::npos) {
168 error_code = validation_errors::MISSING_DESCRIPTION;
169 }
170 else if (issue.message.find("Required directive @COPYRIGHT") != std::string::npos) {
171 error_code = validation_errors::MISSING_COPYRIGHT;
172 }
173
174 ValidationException validation_error(issue.message, error_code, issue.severity);
175 throw std::runtime_error(validation_error.formatted_message());
176 }
177 }
178
179 // After reporting validation issues, report any parser errors
180 if (parser_error) {
181 throw std::runtime_error(*parser_error);
182 }
183
184 // If we got here, we have valid nodes, so compile the query
185 QueryCompiler compiler;
186 for (const auto& node : nodes) {
187 node->accept(compiler);
188 }
189
190 return compiler.get_compiled_query();
191}
